// deno_main.mjs
let msgCount = 100000;
console.log("Sending", msgCount, " messages ...");

let worker = new Worker(new URL("./deno_worker.js", import.meta.url).href, {
  type: "module",
});

const t0 = performance.now();

worker.onmessage = (msg) => {
  if (msg.data >= msgCount) {
    worker.terminate();
    const t1 = performance.now();
    console.log(`Call to postmessage took ${t1 - t0} milliseconds.`);
    Deno.exit();
  } else {
    worker.postMessage(msg.data + 1);
  }
};

worker.postMessage(0);
